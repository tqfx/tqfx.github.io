generate for quintic polynomial trajectory

\begin{aligned} \left\{\begin{array}{l} t=t_{1}-t_{0}\\ q=q_{1}-q_{0}\\ k_{0}=q_{0}\\ k_{1}=v_{0}\\ k_{2}=\cfrac{a_{0}}{2}\\ k_{3}=\cfrac{\left(a_{1}-3\,a_{0}\right)\,t^2+\left(-12\,v_{0}-8\,v_{1}\right)\,t+20\,q}{2\,t^3}\\ k_{4}=\cfrac{\left(3\,a_{0}-2\,a_{1}\right)\,t^2+\left(16\,v_{0}+14\,v_{1}\right)\,t-30\,q}{2\,t^4}\\ k_{5}=\cfrac{\left(a_{1}-a_{0}\right)\,t^2+\left(-6\,v_{0}-6\,v_{1}\right)\,t+12\,q}{2\,t^5} \end{array}\right. \end{aligned}
